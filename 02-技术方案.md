# 02 - 技术方案（方向 A）

> Project DOUYIN-FX ｜ v0.1（方向 A：命令行工作流脚本 + 复用 IDE skill）
> 文档状态：草案

## 0. 与上一版的差异

| | 上一版（已废） | 方向 A（本版） |
|---|---|---|
| UI | Gradio Web UI | **无 UI**，命令行脚本 + IDE chat 触发 |
| 模型推理代码 | 自己用 diffusers + torch 跑 SDXL Turbo | **复用 IDE 的"多模态内容生成"skill**，不写推理代码 |
| 代码量 | ~500 行（Gradio + 推理 + 后处理） | ~150 行（仅后处理 + 打包 + 流水线编排） |
| 核心动作触发 | 浏览器点按钮 | 在 IDE 里一句话 / 命令行一句话 |
| 模型权重管理 | 自己下载、缓存、镜像 fallback | skill 内部处理，工具不管 |
| v0.1 落地周期 | ~1 周 | **2-3 天** |

## 1. 设计原则

- **不造轮子**：模型推理交给 skill，抠图交给 rembg，工具只做"把上下游粘起来 + 抖音 EC 规范化 + 合规提示"。
- **脚本优先**：每个特效场景（S1/S2/S3）一个 workflow 脚本，命令行可跑，IDE 内 chat 也可触发。
- **配方化**：每个 workflow = 一份"配方"（prompt 模板 + 参数 + 后处理步骤），可独立调优、独立运行。
- **沉淀为 skill**：v0.1 跑通后，把工作流沉淀为 IDE skill（详见 `05-GitHub调研与skill计划.md` §5），下次"做一个 X 主题特效"一句话触发。
- **单机单用户**：不考虑并发 / 权限 / 多人。

### 5.1 `lib/skill_bridge.py`

桥接 IDE 的"多模态内容生成"skill。两种模式：

```python
"""
两种调用路径：
1. IDE chat 模式：脚本本身被 IDE agent 调用，agent 直接调 skill；
   workflow 脚本里只声明"我需要 N 张关于 X 的图"，由 agent 拼接到 skill 调用。
2. 命令行模式：脚本独立运行，通过 skill 的 CLI / API 入口调用。
   v0.1 这个入口若不可用，走 §6 兜底（本地 SDXL Turbo）。
"""

from pathlib import Path
from typing import Literal

ImageKind = Literal["text2img", "img2img", "img2video"]


def request_images(
    kind: ImageKind,
    prompt: str,
    *,
    n: int = 4,
    size: tuple[int, int] = (1024, 1024),
    seed: int | None = None,
    output_dir: Path,
    extra: dict | None = None,
) -> list[Path]:
    """请求 N 张候选图，落盘后返回路径列表。

    实现策略（按优先级）：
    1. 检测当前是否运行在 IDE agent 上下文（环境变量 / sentinel 文件） → 输出
       一份"skill 调用请求"到 stdout，由上层 agent 接管；
    2. 否则尝试调用 skill 的 CLI 入口（具体命令在 v0.1 跑通后补，作为
       05 §3 GitHub 调研产物的一部分）；
    3. 都失败 → 走 §6 本地兜底（diffusers + mps），需安装 [local] extras。
    """
    raise NotImplementedError("v0.1 实现：先做策略 1+3，策略 2 等 skill 入口确认")
```

> v0.1 实现先做策略 1（在 IDE 里跑）+ 策略 3（命令行兜底），策略 2 的具体命令等 GitHub 调研（详见 05 文档）后补。

### 5.3 `lib/normalize.py`

```python
import re
from PIL import Image

def normalize_for_ec(
    img: Image.Image,
    target_size: tuple[int, int] = (1024, 1024),
) -> Image.Image:
    """按 Effect Creator 资产规范缩放 + 转 RGBA。"""
    if img.size != target_size:
        img = img.resize(target_size, Image.Resampling.LANCZOS)
    if img.mode != "RGBA":
        img = img.convert("RGBA")
    return img

def safe_filename(s: str, max_len: int = 64) -> str:
    """转 ASCII / 去空格，符合 EC 资产命名习惯。"""
    return re.sub(r"[^a-zA-Z0-9_-]", "_", s)[:max_len]
```

### 5.4 `lib/packager.py`

```python
import json, zipfile
from datetime import datetime
from pathlib import Path
from pydantic import BaseModel, Field

class GenerationInfo(BaseModel):
    backend: str                    # "skill_multimodal_gen" / "local_sdxl_turbo"
    model_id: str | None = None
    prompt: str
    negative_prompt: str = ""
    seed: int = -1
    extra: dict = Field(default_factory=dict)

class AssetItem(BaseModel):
    filename: str
    kind: str                       # "face_texture" / "sticker_frame" / "lut"
    size: tuple[int, int] | None = None
    generation: GenerationInfo
    postprocess: list[str] = Field(default_factory=list)

class PackMeta(BaseModel):
    tool_version: str = "0.1.0"
    project_name: str
    created_at: str = Field(default_factory=lambda: datetime.now().isoformat())
    scenario: str                   # "S1" / "S2" / "S3"
    ai_generated: bool = True
    items: list[AssetItem]


def pack_project(project_dir: Path, meta: PackMeta) -> Path:
    """把 processed/ 打成 zip + 写 meta.json。"""
    meta_path = project_dir / "meta.json"
    meta_path.write_text(meta.model_dump_json(indent=2), encoding="utf-8")

    zip_path = project_dir / f"{meta.project_name}_pack.zip"
    with zipfile.ZipFile(zip_path, "w", zipfile.ZIP_DEFLATED) as z:
        for f in (project_dir / "processed").iterdir():
            z.write(f, arcname=f.name)
        z.write(meta_path, arcname="meta.json")
    return zip_path
```

### 5.5 `lib/safety.py`

```python
COMPLIANCE_ITEMS = [
    "不涉及真实政治人物 / 公众人物 / 名人脸部特征",
    "不涉及色情、露点、性暗示",
    "不涉及血腥、自残、暴力、恐怖元素",
    "不涉及他人肖像 / 商标 / 受版权保护 IP",
    "不涉及政治敏感、民族 / 宗教 / 性别歧视",
    "不涉及违法物品（毒品 / 武器 / 赌博）",
    "已确认按抖音指引在提审时标注 AI 生成",
]

def render_checklist_text() -> str:
    """返回一段控制台 / chat 用的文案，让我自己逐项确认。"""
    lines = ["合规自检清单（请逐项确认 y/n）："]
    for i, item in enumerate(COMPLIANCE_ITEMS, 1):
        lines.append(f"  [{i}] {item}")
    return "\n".join(lines)

def interactive_confirm() -> bool:
    """命令行交互：全部 y 才返回 True。"""
    print(render_checklist_text())
    for i, item in enumerate(COMPLIANCE_ITEMS, 1):
        ans = input(f"[{i}] 确认？(y/n): ").strip().lower()
        if ans != "y":
            return False
    return True
```

### 5.6 `workflows/s1_face_paint.py`

```python
"""S1：抖音面部贴图素材流水线（v0.1 必做）。

用法：
  python -m workflows.s1_face_paint \
    --prompt "cyberpunk metal mask" \
    --project cyberpunk-mask-v1 \
    --n 4

或者在 IDE chat 里："用 S1 流水线做一个赛博朋克金属面具的特效"
"""

import click
from pathlib import Path
from datetime import date

from lib.skill_bridge import request_images
from lib.matting import remove_background
from lib.normalize import normalize_for_ec, safe_filename
from lib.packager import pack_project, PackMeta, AssetItem, GenerationInfo
from lib.safety import interactive_confirm
from PIL import Image


PROMPT_TEMPLATE = Path("prompts/face_paint_template.md").read_text(encoding="utf-8")


@click.command()
@click.option("--prompt", required=True, help="主题描述，例如 'cyberpunk metal mask'")
@click.option("--project", required=True, help="项目名，用作输出目录与 zip 命名")
@click.option("--n", default=4, help="出几张候选")
@click.option("--seed", default=None, type=int)
@click.option("--output-root", default="output", type=click.Path())
def main(prompt: str, project: str, n: int, seed: int | None, output_root: str):
    out_dir = Path(output_root) / f"{date.today().isoformat()}-{safe_filename(project)}"
    (out_dir / "candidates").mkdir(parents=True, exist_ok=True)
    (out_dir / "selected").mkdir(exist_ok=True)
    (out_dir / "processed").mkdir(exist_ok=True)

    # 1. 拼 prompt
    final_prompt = PROMPT_TEMPLATE.replace("{{THEME}}", prompt)
    print(f"[1/6] Prompt: {final_prompt[:120]}...")

    # 2. 调 skill 出 N 张候选
    print(f"[2/6] 请求 {n} 张候选图...")
    candidates = request_images(
        kind="text2img",
        prompt=final_prompt,
        n=n,
        size=(1024, 1024),
        seed=seed,
        output_dir=out_dir / "candidates",
    )

    # 3. 让我选一张
    print(f"[3/6] 候选已生成在 {out_dir / 'candidates'}")
    for i, p in enumerate(candidates, 1):
        print(f"  [{i}] {p.name}")
    idx = int(input("选哪张？(输入序号): ").strip())
    selected = candidates[idx - 1]
    selected_dst = out_dir / "selected" / selected.name
    selected_dst.write_bytes(selected.read_bytes())

    # 4. 抠背景
    print("[4/6] 抠透明背景...")
    matted = remove_background(Image.open(selected_dst))

    # 5. 规范化
    print("[5/6] 规范化为 EC 资产格式...")
    normalized = normalize_for_ec(matted, (1024, 1024))
    final_name = safe_filename(f"{project}_face_paint_main") + ".png"
    final_path = out_dir / "processed" / final_name
    normalized.save(final_path, "PNG")

    # 6. 合规自检 → 打包
    print("[6/6] 合规自检...")
    if not interactive_confirm():
        print("自检未通过，未生成 zip。")
        return

    meta = PackMeta(
        project_name=project,
        scenario="S1",
        items=[AssetItem(
            filename=final_name,
            kind="face_texture",
            size=(1024, 1024),
            generation=GenerationInfo(
                backend="skill_multimodal_gen",
                prompt=final_prompt,
                seed=seed if seed is not None else -1,
            ),
            postprocess=["rembg", "resize_1024x1024", "rgba"],
        )],
    )
    zip_path = pack_project(out_dir, meta)
    print(f"\n完成：{zip_path}")
    print(f"下一步：把 zip 解压后的 PNG 拖进 Effect Creator。")


if __name__ == "__main__":
    main()
```
